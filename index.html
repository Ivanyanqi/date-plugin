live quick start

最小接入

下面这段代码会跟随你当前试玩区的配置同步更新，方便直接观察接入形态会怎么变化。

<link rel="stylesheet" href="date.style.css">
<script type="module">
  import { DatePicker } from "./src/index.js";

  const picker = new DatePicker(
    document.querySelector("#start"),
    {
      locale: "zh-CN",
      format: "YYYY-MM-DD",
      showToday: true,
      showClear: true
    }
  );

  await picker.ready;
</script>
